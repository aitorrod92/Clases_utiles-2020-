    /**
     * Obtiene un string con el contenido de un archivo de texto externo (no contenido en el JAR).
     *
     * @param archivo Archivo que se quiere leer.
     * @return String con el contenido del archivo.
     * @throws FileNotFoundException Si el archivo no existe.
     * @throws IOException Si el archivo no se puede leer.
     */
    private String ObtenerTexto(File archivo) throws FileNotFoundException, IOException {
        FileReader fr = new FileReader(archivo);
        BufferedReader br = new BufferedReader(new FileReader(archivo));
        String string = "";
        String line = null;
        while ((line = br.readLine()) != null) {
            string = string.concat(line + "\n");
        }
        br.close();
        fr.close();
        return string;
    }
